總死亡數最高的國家

SELECT
    "國家",
    MAX("總死亡數") AS 總死亡數
FROM
    world
GROUP BY
    "國家"
ORDER BY
    總死亡數 DESC
LIMIT 1;

台灣哪個月死亡人數最多人


SELECT
    DATE_TRUNC('month', "日期") AS 死亡月份,
    SUM("新增死亡數") AS 月新增死亡數
FROM
    world
WHERE
    "國家" = '台灣'
GROUP BY
    死亡月份
ORDER BY
    月新增死亡數 DESC
LIMIT 1;
